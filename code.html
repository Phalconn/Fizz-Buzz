//get values from the user. we need to get the fizz and the buzz value.
function getValues(){

    //get the user values from the page
    let fizzValue = document.getElementById("fizzValue").value;
    let buzzValue = document.getElementById("buzzValue").value;

    //parse for numbers
    fizzValue = parseInt(fizzValue);
    buzzValue = parseInt(buzzValue);

    //check to make sure they are integers
    if(Number.isInteger(fizzValue) && Number.isInteger(buzzValue)) {

        //call the fizzBuzz
        let fbArray = fizzBuzz(fizzValue,buzzValue);
        //call displayData and write the values to the screen
        displayData(fbArray);

    } else {
        alert("Error: Program is expecting an integer");
    }


}
//do fizz buzz
function fizzBuzz(fizzValue, buzzValue) 
    {

    let returnArray = [];

    
    //loop from 1 to 100
    for (let i = 1; i <= 100; i++) {
        //check to see if divisible by both (3 and 5)
        //check to see if divisible by fizz value (3)
        //check to see if divisible by fizz value (5)
            if(i % fizzValue == 0 && i % buzzValue == 0){
                returnArray.push('FizzBuzz');
            }else if(i % fizzValue == 0) {
                returnArray.push('Fizz');
            }else if(i % buzzValue ==0) {
                returnArray.push('Buzz');
            }else{
                returnArray.push(i);
            }

    }
    return returnArray;  
}

function displayData(fbArray){

    //get table body element from the page
    let tableBody = document.getElementById("results");

    //get the template row
    let templateRow = document.getElementById("fbTemplate");


    //clear table first
    tableBody.innerHTML = "";

    for (let index = 0; index < fbArray.length; index += 5) {

        let tableRow = document.importNode(templateRow.content, true);

        //grab just the  and put them into an array
        let rowCols = tableRow.querySelectorAll("td");
        rowCols[1].classList.add(fbArray[index]);
        rowCols[0].textContent = fbArray[index];
        
        rowCols[2].classList.add(fbArray[index + 1]);
        rowCols[1].textContent = fbArray[index+1];

        rowCols[2].classList.add(fbArray[index+ 2]);
        rowCols[2].textContent = fbArray[index+2];

        rowCols[3].classList.add(fbArray[index+ 3]);
        rowCols[3].textContent = fbArray[index+3];

        rowCols[4].classList.add(fbArray[index+ 4]);
        rowCols[4].textContent = fbArray[index+4];
        
        tableBody.appendChild(tableRow);


    }

    //add all rows to table

}

The code is structured with several functions.

Get the values from the user

First off we start by writing a function to get values from the user to determine what to display. The variable getValues is declared and in consists of two several functions.

The let statement is declared using two variables, fizzValue and buzzValue to obtain the users input.

Next, we use a method to parse for numbers. The parseInt() function parses a string argument and returns an integer of the specified radix (the base in mathematical numeral systems).

Then we have to validate if the user has entered integers by using an If statement(Number.isInteger(fizzValue) && Number.isInteger(buzzValue)) {}.

By using a let statement with a variable of fbArray, we are able to call fizzBuzz by calling a function and passing our variables fizzValue and buzzValue as parameters in the function.

Do Fizz Buzz

Next, we write a function that will make the program display “Fizz Buzz”. Simply creating a for loop with an if/else statement to iterate from 1 to 100. Once that has been done we write another function to return the array. We did so by declaring returnArray.push(i).

Here, the program checks for the following:

1. Check tho see if its divisible by both (3 and 5). If so push FizzBuzz in the array

2. Check to see if divisible by Fizz Value(3). If so push “fizz” into the array and not a number

3. Check to see if divisible by Buzz Value(5) If so push “buzz” into the array and not a number

4.If none then push number into the array

Call display data and write the values to the screen

After doing so, we need the application to display the data by declaring a function called displayData with the parameter fArray

Get the table body element from the page by using a document fragment.

We are also going to use a template literal which allows the use of multi-line strings and string interpolation features with them.

In the site.js file we declare a let statement with a variable of tableBody and using document.getElementById in order to retrieve the “results” from our inner HTML. Our inner HTML of element “results” is set with our a template element with an id of fbTemplate that consists of table rows and table data elements.

Get the template row

We get the template row by also declaring a let statement of templateRow and. Use a DOM (document.getElementByID(“fbTemplate”)) and pass in the fbTemplate element from our inner HTML.

Clearing the table is crucial too. If we have a hundred numbers on display, we don’t want the program to add another hundred numbers by declaring tableBody.InnerHTML = “”;.

Add all rows to the table

Using a for loop with a combination of document fragments, the program enables us to add table rows and table data to display the results. Since the for loop is going through numbers one to hundred, we should write a function ta add rows and columns as long as we are less than the length of the array and stops at the index of [99] so we don’t go out of bounds.
